Arduino Keypad Lock Code

        
#include <Servo.h>
#include <Keypad.h>

#define PASSWORD "1234"
Servo lockServo; 
const byte ROWS = 4; 
const byte COLS = 4; 
char keys[ROWS][COLS] = {
    {'1', '2', '3', 'A'},
    {'4', '5', '6', 'B'},
    {'7', '8', '9', 'C'},
    {'*', '0', '        
};
byte rowPins[ROWS] = {9, 8, 7, 6}; 
byte colPins[COLS] = {5, 4, 3, 2}; 

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);
String inputPassword = "";

void setup() {
    lockServo.attach(10); 
}

void loop() {
    char key = keypad.getKey();
    if (key) {
        if (key == '#') {
            if (inputPassword == PASSWORD) {
                lockServo.write(90); 
                delay(5000); 
                lockServo.write(0); 
            }
            inputPassword = ""; 
        } else {
            inputPassword += key; 
        }
    }
}
